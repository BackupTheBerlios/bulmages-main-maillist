CMakeLists.txt

   GETTEXT_CREATE_TRANSLATIONS(${CMAKE_CURRENT_BINARY_DIR}/${plugin}.pot
                                ALL LANGUAGES es_ES.UTF-8 ca_ES.UTF-8)
