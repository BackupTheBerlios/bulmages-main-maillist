/// DELEGADO: constructor
DesplegableProveedores::DesplegableProveedores
( BlMainCompany *company, QObject *padre )
: QItemDelegate(padre), m_company(company)
{
}


/// DELEGADO: visualización
void DesplegableProveedores::paint(QPainter *pintor
, const QStyleOptionViewItem &vis, const QModelIndex &elemento) const
{
   QItemDelegate::paint(pintor, vis, elemento);
}


/// DELEGADO: creación del widget de edición
QWidget *DesplegableProveedores::createEditor(QWidget *padre,
const QStyleOptionViewItem &vis, const QModelIndex &elemento) const
{
   BlComboBox *desplegable = new BlComboBox(padre->parentWidget());
   desplegable->setMainCompany(m_company);
   desplegable->setQuery("SELECT id, nombre FROM gen.proveedor ORDER BY 
nombre");
   desplegable->setFieldId("id");
   desplegable->m_valores["nombre"] = "";
   desplegable->setAllowNull(true);

/// Si en vez de cargar una consulta, inserto elementos uno a uno, todo 
funciona bien: no hay fallo de segmentación
/// ((BlComboBox *)editor)->insertItem(0, "hola");

   return desplegable;
}


/// DELEGADO: poner los datos en el widget de edición
void DesplegableProveedores::setEditorData(QWidget *editor, const QModelIndex 
&elemento) const
{
   /// Esto provoca un fallo de segmentación
   ((BlComboBox *)editor)->setId(elemento.model()->data(elemento).toString());
}


/// DELEGADO: Poner los cambios en la base de datos
void DesplegableProveedores::setModelData(QWidget *editor,
QAbstractItemModel *modelo, const QModelIndex &elemento) const
{
   modelo->setData(elemento, ((BlComboBox *)editor)->id());
}


/// DELEGADO: actualizar posición y tamaño del widget de edición
void DesplegableProveedores::updateEditorGeometry(QWidget *editor, const 
QStyleOptionViewItem &vis, const QModelIndex &elemento) const
{
   editor->setGeometry(vis.rect);
}
